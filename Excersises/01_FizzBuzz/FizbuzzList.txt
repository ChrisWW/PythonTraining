#Stwórz listę result, która będzie zawierać liczby od 1 do 100 (włącznie).
#W liście result liczby:
#- podzielne przez 3 zastąp stringiem 'Fizz'
#- podzielne przez 5 zastąp stringiem 'Buzz'
#- podzielne jednocześnie przez 3 i przez 5 zastąp stringiem 'FizzBuzz'

result = list(i for i in range(1, 101))

numbers = [int(x) for x in result]
mylist =[]

for m in result:
    if m % 3 == 0:
        mylist.append(str("Fizz"))
    if m % 5 == 0:
        mylist.append(str("Buzz"))
    if m % 15 == 0:
        mylist.append(str("FizzBuzz"))
    else:
        mylist.append(str(m))

print(mylist)
